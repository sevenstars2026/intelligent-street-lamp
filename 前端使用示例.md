## 📱 React 使用示例

### 配置API基础URL
```javascript
// src/api/config.js
export const API_BASE_URL = 'http://localhost:3000/api';
```

### 1. 控制路灯开关
```javascript
// src/api/device.js
async function controlLight(deviceId, command) {
  const response = await fetch(`${API_BASE_URL}/devices/${deviceId}/control`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({ command }) // "on" 或 "off"
  });
  
  const data = await response.json();
  
  if (data.code === 200) {
    console.log('控制成功:', data.data);
    return data.data;
  } else {
    console.error('控制失败:', data.message);
    throw new Error(data.message);
  }
}

// 使用
controlLight('lamp_001', 'on')
  .then(result => {
    alert('路灯已打开');
  })
  .catch(error => {
    alert('操作失败: ' + error.message);
  });
```

### 2. 批量控制路灯
```javascript
async function batchControl(deviceIds, command) {
  const response = await fetch(`${API_BASE_URL}/devices/batch-control`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({ 
      deviceIds: deviceIds,  // ['lamp_001', 'lamp_002']
      command: command       // "on" 或 "off"
    })
  });
  
  return await response.json();
}

// 使用：同时控制多个路灯
batchControl(['lamp_001', 'lamp_002', 'lamp_003'], 'off')
  .then(result => {
    console.log('批量控制结果:', result.data);
  });
```

### 8. 获取历史光照数据
```javascript
async function getLightHistory(deviceId, startTime, endTime, aggregation = 'raw') {
  const params = new URLSearchParams({
    startTime: startTime,  // '2026-06-01T00:00:00Z'
    endTime: endTime,      // '2026-06-30T23:59:59Z'
    aggregation: aggregation // 'raw', 'hourly', 'daily'
  });
  
  const response = await fetch(
    `${API_BASE_URL}/devices/${deviceId}/light-history?${params}`
  );
  const data = await response.json();
  return data.data;
}

// 使用：获取最近7天的每小时平均数据
const endTime = new Date().toISOString();
const startTime = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000).toISOString();

getLightHistory('lamp_001', startTime, endTime, 'hourly')
  .then(data => {
    console.log('历史数据:', data);
    // 可以用这些数据画图表
  });
```

## 🎨 Vue 使用示例

### 配置axios
```javascript
// src/api/axios.js
import axios from 'axios';

const api = axios.create({
  baseURL: 'http://localhost:3000/api',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
});

// 响应拦截器
api.interceptors.response.use(
  response => {
    const { code, message, data } = response.data;
    if (code === 200) {
      return data;
    } else {
      return Promise.reject(new Error(message));
    }
  },
  error => {
    return Promise.reject(error);
  }
);

export default api;
```

### API封装
```javascript
// src/api/device.js
import api from './axios';

export default {
  // 控制路灯
  control(deviceId, command) {
    return api.post(`/devices/${deviceId}/control`, { command });
  },
  
  // 批量控制
  batchControl(deviceIds, command) {
    return api.post('/devices/batch-control', { deviceIds, command });
  },
  
  // 获取阈值
  getThreshold(deviceId) {
    return api.get(`/devices/${deviceId}/threshold`);
  },
  
  // 设置阈值
  setThreshold(deviceId, lightThresholdOn, lightThresholdOff) {
    return api.post(`/devices/${deviceId}/threshold`, {
      lightThresholdOn,
      lightThresholdOff
    });
  },
  
  // 获取模式
  getMode(deviceId) {
    return api.get(`/devices/${deviceId}/mode`);
  },
  
  // 切换模式
  setMode(deviceId, mode) {
    return api.put(`/devices/${deviceId}/mode`, { mode });
  },
  
  // 控制历史
  getControlLogs(deviceId, page = 1, pageSize = 20) {
    return api.get(`/devices/${deviceId}/control-logs`, {
      params: { page, pageSize }
    });
  },
  
  // 历史光照数据
  getLightHistory(deviceId, startTime, endTime, aggregation = 'raw') {
    return api.get(`/devices/${deviceId}/light-history`, {
      params: { startTime, endTime, aggregation }
    });
  },
  
  // 设备统计
  getStatistics(deviceId) {
    return api.get(`/devices/${deviceId}/statistics`);
  }
};
```

### Vue组件使用
```vue
<template>
  <div>
    <button @click="turnOn">开灯</button>
    <button @click="turnOff">关灯</button>
    <div>当前模式: {{ mode }}</div>
    <button @click="toggleMode">切换模式</button>
  </div>
</template>

<script>
import deviceApi from '@/api/device';

export default {
  data() {
    return {
      deviceId: 'lamp_001',
      mode: 'auto'
    };
  },
  
  mounted() {
    this.loadMode();
  },
  
  methods: {
    async turnOn() {
      try {
        await deviceApi.control(this.deviceId, 'on');
        this.$message.success('开灯成功');
      } catch (error) {
        this.$message.error('操作失败: ' + error.message);
      }
    },
    
    async turnOff() {
      try {
        await deviceApi.control(this.deviceId, 'off');
        this.$message.success('关灯成功');
      } catch (error) {
        this.$message.error('操作失败: ' + error.message);
      }
    },
    
    async loadMode() {
      try {
        const data = await deviceApi.getMode(this.deviceId);
        this.mode = data.mode;
      } catch (error) {
        console.error('获取模式失败:', error);
      }
    },
    
    async toggleMode() {
      try {
        const newMode = this.mode === 'auto' ? 'manual' : 'auto';
        await deviceApi.setMode(this.deviceId, newMode);
        this.mode = newMode;
        this.$message.success('模式切换成功');
      } catch (error) {
        this.$message.error('切换失败: ' + error.message);
      }
    }
  }
};
</script>
```

## 📊 完整响应格式

所有接口都返回统一格式：

```javascript
{
  "code": 200,        // 状态码：200成功，400/404/500错误
  "message": "成功",  // 提示信息
  "data": { ... }     // 实际数据
}
```

### 错误处理示例
```javascript
async function apiCall() {
  try {
    const response = await fetch('...');
    const result = await response.json();
    
    if (result.code === 200) {
      // 成功
      return result.data;
    } else {
      // 业务错误
      throw new Error(result.message);
    }
  } catch (error) {
    // 网络错误或其他错误
    console.error('请求失败:', error);
    throw error;
  }
}
```

## 🔧 常见问题

**Q: 请求失败怎么办？**
```bash
# 检查后端是否在运行
curl http://localhost:3000/api/health

# 如果失败，重新启动后端
cd task3-backend
npm run dev
```

**Q: 端口被占用？**
```bash
# 修改端口（在task3-backend/.env文件）
PORT=3001
```

**Q: 跨域问题？**  
后端已配置CORS，如果还有问题，检查前端请求URL是否正确。
